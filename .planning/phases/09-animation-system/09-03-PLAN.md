    ```typescript
    import React from 'react';
    import { Spinner } from '@inkjs/ui';

    type SpinnerContext = 'narration' | 'npc_dialogue' | 'combat';

    type SceneSpinnerProps = {
      readonly context: SpinnerContext;
    };

    const SPINNER_LABELS: Record<SpinnerContext, readonly string[]> = {
      narration: ['命运之轮转动中...', '史官正在记录...'],
      npc_dialogue: ['正在思考...'],
      combat: ['攻击展开...', '局势变化...'],
    };

    export function SceneSpinner({ context }: SceneSpinnerProps): React.ReactNode {
      const labels = SPINNER_LABELS[context];
      // Select a random label -- use useState to keep it stable across re-renders
      const [label] = React.useState(() =>
        labels[Math.floor(Math.random() * labels.length)]!
      );
      return <Spinner type="dots" label={label} />;
    }
    ```

    ```typescript
    import type { ToastData } from '../hooks/use-toast';

    type ScenePanelProps = {
      readonly lines: readonly string[];
      readonly streamingText?: string;
      readonly isStreaming?: boolean;
      readonly showSpinner?: boolean;
      readonly spinnerContext?: 'narration' | 'npc_dialogue' | 'combat';
      readonly toast?: ToastData | null;
      readonly isDimmed?: boolean;
      readonly isSpinnerDimming?: boolean;
    };
    ```

    The `isSpinnerDimming` prop supports D-07's spinner-to-streaming transition. When true, the spinner renders with `dimColor` to create a dim-out effect before streaming content replaces it. GameScreen (Plan 05) manages the state machine that drives this prop.

    Render order inside the `<Box flexDirection="column" flexGrow={1} paddingX={1}>`:
    1. **Toast** (at top, before content): `{toast && <ToastBanner toast={toast} />}`
    2. **Content** (existing lines + streaming, with dimColor support):
       - If `showSpinner` is true AND no streamingText yet: render `<SceneSpinner context={spinnerContext ?? 'narration'} />`, wrapped in `<Text dimColor={isSpinnerDimming}>` when dimming
       - Else: render existing content (lines + streaming text)
       - Apply `dimColor={isDimmed}` directly to each Text element that renders narration lines

    **Revised render structure:**
    ```tsx
    export function ScenePanel({ lines, streamingText, isStreaming, showSpinner, spinnerContext, toast, isDimmed, isSpinnerDimming }: ScenePanelProps): React.ReactNode {
      return (
        <Box flexDirection="column" flexGrow={1} paddingX={1}>
          {toast && <ToastBanner toast={toast} />}
          {showSpinner && !streamingText ? (
            <Text dimColor={isSpinnerDimming}>
              <SceneSpinner context={spinnerContext ?? 'narration'} />
            </Text>
          ) : lines.length === 0 && !streamingText ? (
            <Text dimColor>周围一片寂静，什么也没有发生。</Text>
          ) : (
            <>
              {lines.map((line, i) => (
                <Text key={i} dimColor={isDimmed}>{line}</Text>
              ))}
              {isStreaming && (
                <Text dimColor={isDimmed}>
                  {streamingText}
                  <Text dimColor>...</Text>
                </Text>
              )}
            </>
          )}
        </Box>
      );
    }
    ```

    This preserves all existing ScenePanel callers (new props are optional, defaulting to undefined/false).
