    ```typescript
    import React from 'react';
    import { Spinner } from '@inkjs/ui';

    type SpinnerContext = 'narration' | 'npc_dialogue' | 'combat';

    type SceneSpinnerProps = {
      readonly context: SpinnerContext;
    };

    const SPINNER_LABELS: Record<SpinnerContext, readonly string[]> = {
      narration: ['命运之轮转动中...', '史官正在记录...'],
      npc_dialogue: ['正在思考...'],
      combat: ['攻击展开...', '局势变化...'],
    };

    export function SceneSpinner({ context }: SceneSpinnerProps): React.ReactNode {
      const labels = SPINNER_LABELS[context];
      // Select a random label — use useState to keep it stable across re-renders
      const [label] = React.useState(() =>
        labels[Math.floor(Math.random() * labels.length)]!
      );
      return <Spinner type="dots" label={label} />;
    }
    ```

    Modify ScenePanel to accept new optional props and render the new slots. Expand `ScenePanelProps`:

    ```typescript
    import type { ToastData } from '../hooks/use-toast';

    type ScenePanelProps = {
      readonly lines: readonly string[];
      readonly streamingText?: string;
      readonly isStreaming?: boolean;
      readonly showSpinner?: boolean;
      readonly spinnerContext?: 'narration' | 'npc_dialogue' | 'combat';
      readonly toast?: ToastData | null;
      readonly isDimmed?: boolean;
    };
    ```

    Render order inside the `<Box flexDirection="column" flexGrow={1} paddingX={1}>`:
    1. **Toast** (at top, before content): `{toast && <ToastBanner toast={toast} />}`
    2. **Content** (existing lines + streaming, wrapped in FadeWrapper):
       - If `showSpinner` is true AND no streamingText yet: render `<SceneSpinner context={spinnerContext ?? 'narration'} />`
       - Else: render existing content (lines + streaming text)
       - Wrap the content block in `<FadeWrapper isDimmed={isDimmed ?? false}>...</FadeWrapper>` — but only if `isDimmed` is true, otherwise render directly (avoid unnecessary wrapper nesting that might affect Ink layout).

    Actually, since FadeWrapper uses `<Text dimColor>`, wrapping a `<Box>` of children inside `<Text>` is invalid in Ink (Text can only contain strings/Text). So instead of FadeWrapper as a wrapper component, apply `dimColor` directly to each Text element. Simpler approach:

    Pass `isDimmed` to the rendering logic and add `dimColor={isDimmed}` to the Text elements that render narration lines:
    ```tsx
    {lines.map((line, i) => <Text key={i} dimColor={isDimmed}>{line}</Text>)}
    ```

    For the streaming text and spinner, apply dimColor similarly.

    **Revised render structure:**
    ```tsx
    export function ScenePanel({ lines, streamingText, isStreaming, showSpinner, spinnerContext, toast, isDimmed }: ScenePanelProps): React.ReactNode {
      return (
        <Box flexDirection="column" flexGrow={1} paddingX={1}>
          {toast && <ToastBanner toast={toast} />}
          {showSpinner && !streamingText ? (
            <SceneSpinner context={spinnerContext ?? 'narration'} />
          ) : lines.length === 0 && !streamingText ? (
            <Text dimColor>周围一片寂静，什么也没有发生。</Text>
          ) : (
            <>
              {lines.map((line, i) => (
                <Text key={i} dimColor={isDimmed}>{line}</Text>
              ))}
              {isStreaming && (
                <Text dimColor={isDimmed}>
                  {streamingText}
                  <Text dimColor>...</Text>
                </Text>
              )}
            </>
          )}
        </Box>
      );
    }
    ```

    This preserves all existing ScenePanel callers (new props are optional, defaulting to undefined/false).

    Remove `FadeWrapper` file if the above approach works — we don't need it as a separate component since Ink's Text nesting rules make wrapping impractical. Instead, `isDimmed` is applied directly. Keep the FadeWrapper file but note it's unused (or delete it in this task and update Task 1's files list). Actually, keep it — it may be useful for title screen subtitle fade or other contexts where single-Text-child wrapping works.
