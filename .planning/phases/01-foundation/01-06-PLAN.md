```typescript
import { z } from 'zod';
import type { Store } from './create-store';
import { PlayerStateSchema, type PlayerState } from './player-store';
import { SceneStateSchema, type SceneState } from './scene-store';
import { CombatStateSchema, type CombatState } from './combat-store';
import { GameStateSchema, type GameState } from './game-store';

interface StoreEntry<T> {
  store: Store<T>;
  schema: z.ZodType<T>;
}

export interface Serializer {
  snapshot(): string;
  restore(json: string): void;
}

const SaveDataSchema = z.object({
  version: z.literal(1),
  timestamp: z.string(),
  player: PlayerStateSchema,
  scene: SceneStateSchema,
  combat: CombatStateSchema,
  game: GameStateSchema,
});

export function createSerializer(stores: {
  player: Store<PlayerState>;
  scene: Store<SceneState>;
  combat: Store<CombatState>;
  game: Store<GameState>;
}): Serializer {
  return {
    snapshot(): string {
      const data = {
        version: 1 as const,
        timestamp: new Date().toISOString(),
        player: stores.player.getState(),
        scene: stores.scene.getState(),
        combat: stores.combat.getState(),
        game: stores.game.getState(),
      };
      return JSON.stringify(data);
    },

    restore(json: string): void {
      let raw: unknown;
      try {
        raw = JSON.parse(json);
      } catch {
        throw new Error('Invalid save data: malformed JSON');
      }

      const result = SaveDataSchema.safeParse(raw);
      if (!result.success) {
        const firstError = result.error.errors[0];
        throw new Error(
          `Invalid save data: ${firstError?.path.join('.')} — ${firstError?.message}`,
        );
      }

      const data = result.data;
      stores.player.setState(() => data.player);
      stores.scene.setState(() => data.scene);
      stores.combat.setState(() => data.combat);
      stores.game.setState(() => data.game);
    },
  };
}
```

Note: The `setState(() => data.player)` pattern works because immer's produce with a recipe that returns a value replaces the entire state. However, the create-store setState signature is `(recipe: (draft: T) => void) => void` which expects a mutation recipe, not a return value. The serializer needs a special approach:
- Option A: Add a `replaceState(newState: T)` method to Store
- Option B: Use `Object.assign(draft, data.player)` in the recipe to overwrite all fields
- Option C: Modify createStore to detect when recipe returns a value

Choose Option B (`Object.assign(draft, validatedData)`) as it works with the existing immer recipe API without modifying createStore.

```typescript
import { createCommandParser, type CommandParser } from './input/command-parser';
import { routeInput, type InputResult } from './input/input-router';
import { resolveNormalCheck } from './engine/adjudication';
import { rollD20 } from './engine/dice';
import { calculateDamage } from './engine/damage';
import { playerStore } from './state/player-store';
import { sceneStore } from './state/scene-store';
import { gameStore } from './state/game-store';
import { eventBus } from './events/event-bus';
import type { GameAction } from './types/game-action';
import type { CheckResult } from './types/common';
import { ATTRIBUTE_LABELS } from './types/common';

export interface GameLoop {
  processInput: (input: string) => Promise<ProcessResult>;
  getCommandParser: () => CommandParser;
}

export type ProcessResult =
  | { status: 'action_executed'; action: GameAction; checkResult?: CheckResult; narration: string[] }
  | { status: 'help'; commands: string[] }
  | { status: 'clarification'; message: string }
  | { status: 'error'; message: string };

export function createGameLoop(): GameLoop {
  const commandParser = createCommandParser();

  async function processInput(input: string): Promise<ProcessResult> {
    // Step 1: Route input to command parser or NL classifier
    const sceneContext = sceneStore.getState().narrationLines.join(' ');
    const routeResult = await routeInput(input, commandParser, sceneContext);

    if (routeResult.status === 'error') {
      return { status: 'error', message: routeResult.message };
    }
    if (routeResult.status === 'clarification') {
      return { status: 'clarification', message: routeResult.message };
    }

    const action = routeResult.action;

    // Step 2: Handle non-adjudicated actions
    if (action.type === 'help') {
      return {
        status: 'help',
        commands: [
          '/look [target] — 观察周围或特定目标',
          '/go <direction> — 向指定方向移动',
          '/talk <npc> — 与NPC对话',
          '/attack <target> — 攻击目标',
          '/inspect <target> — 检查物品',
          '/use_item <item> — 使用物品',
          '/cast <spell> [target] — 施放法术',
          '/guard — 防御',
          '/flee — 逃跑',
          '/trade <npc> — 与NPC交易',
          '/save [name] — 保存游戏',
          '/help — 显示此帮助',
        ],
      };
    }

    if (action.type === 'look') {
      // Look action: display current scene narration (no check needed)
      const scene = sceneStore.getState();
      return {
        status: 'action_executed',
        action,
        narration: scene.narrationLines,
      };
    }

    // Step 3: Adjudicate action via Rules Engine (placeholder for Phase 1)
    // In Phase 1, demonstrate the Rules Engine by running a check for any action
    const player = playerStore.getState();
    const attributeName = getRelevantAttribute(action.type);
    const attrMod = player.attributes[attributeName] ?? 0;
    const roll = rollD20();
    const dc = 12; // Placeholder DC for Phase 1

    const checkResult = resolveNormalCheck({
      roll,
      attributeName,
      attributeModifier: attrMod,
      skillModifier: 0,
      environmentModifier: 0,
      dc,
    });

    // Step 4: Emit domain event
    eventBus.emit('action_resolved', { action, result: checkResult });

    // Step 5: Update scene narration with check result display (D-20)
    const newNarration = [
      ...sceneStore.getState().narrationLines,
      checkResult.display,
    ];
    sceneStore.setState(draft => {
      draft.narrationLines = newNarration;
    });

    // Step 6: Advance turn
    gameStore.setState(draft => {
      draft.turnCount += 1;
    });

    return {
      status: 'action_executed',
      action,
      checkResult,
      narration: newNarration,
    };
  }

  return { processInput, getCommandParser: () => commandParser };
}

function getRelevantAttribute(actionType: string): 'physique' | 'finesse' | 'mind' {
  switch (actionType) {
    case 'attack':
    case 'guard':
    case 'flee':
      return 'physique';
    case 'inspect':
    case 'use_item':
    case 'trade':
      return 'finesse';
    case 'talk':
    case 'cast':
      return 'mind';
    default:
      return 'physique';
  }
}
```

import { describe, test, expect } from 'bun:test';
import { createCommandParser } from '../input/command-parser';

    expect(go).not.toBeNull();

    expect(talk).not.toBeNull();

    expect(talk!.target).toBe('guard');

  });
});
