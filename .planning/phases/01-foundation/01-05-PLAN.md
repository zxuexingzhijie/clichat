```typescript
import { Command } from 'commander';
import type { GameAction } from '../types/game-action';
import { registerCommands } from './command-registry';

export type CommandParser = {
  parse: (input: string) => GameAction | null;
};

export function createCommandParser(): CommandParser {
  const program = new Command();
  program.exitOverride();
  program.configureOutput({
    writeOut: () => {},
    writeErr: () => {},
  });

  let result: GameAction | null = null;
  const setResult = (action: GameAction) => { result = action; };

  registerCommands(program, setResult);

  return {
    parse(input: string): GameAction | null {
      result = null;
      const cleaned = input.replace(/^\//, '').trim();
      if (!cleaned) return null;
      const args = cleaned.split(/\s+/);
      try {
        program.parse(args, { from: 'user' });
      } catch {
        return null;
      }
      return result;
    },
  };
}
```

Create `src/input/intent-classifier.ts` per D-24/D-25 and RESEARCH.md Pattern 4:
```typescript
import { generateObject } from 'ai';
import { openai } from '@ai-sdk/openai';
import { IntentSchema, type Intent } from '../types/intent';

const INTENT_SYSTEM_PROMPT = `You are an intent classifier for a Chinese fantasy RPG game.
Classify the player's input into a structured game action.
Available actions: move, look, talk, attack, use_item, cast, guard, flee, inspect, trade.
Respond with the action type, target (if any), and your confidence level.
If the input is ambiguous, set confidence below 0.5 and provide your best interpretation.`;

export async function classifyIntent(
  input: string,
  sceneContext: string,
  options?: { maxRetries?: number },
): Promise<Intent> {
  const maxRetries = options?.maxRetries ?? 1;
  let lastError: unknown;

  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      const { object } = await generateObject({
        model: openai('gpt-4o-mini'),
        schema: IntentSchema,
        prompt: `${INTENT_SYSTEM_PROMPT}

Current scene: ${sceneContext}
Player input: ${input}

Classify the player's intent.`,
      });
      return object;
    } catch (err) {
      lastError = err;
    }
  }

  throw new Error(
    `Intent classification failed after ${maxRetries + 1} attempts: ${lastError instanceof Error ? lastError.message : String(lastError)}`,
  );
}
```

Create `src/input/input-router.ts` per D-25/D-39:
```typescript
import type { GameAction } from '../types/game-action';
import type { CommandParser } from './command-parser';
import { classifyIntent } from './intent-classifier';

export type InputResult =
  | { status: 'success'; action: GameAction }
  | { status: 'clarification'; message: string; candidates: GameAction[] }
  | { status: 'error'; message: string };

const CONFIDENCE_THRESHOLD = 0.3;

export async function routeInput(
  input: string,
  commandParser: CommandParser,
  sceneContext: string,
): Promise<InputResult> {
  const trimmed = input.trim();
  if (!trimmed) {
    return { status: 'error', message: '请输入命令或描述你想做的事。' };
  }

  // D-39: Input starting with / is a command
  if (trimmed.startsWith('/')) {
    const action = commandParser.parse(trimmed);
    if (action) {
      return { status: 'success', action };
    }
    return {
      status: 'error',
      message: '未知命令。输入 /help 查看可用命令。',
    };
  }

  // Everything else goes through NL intent classification
  try {
    const intent = await classifyIntent(trimmed, sceneContext);

    // D-27: Low confidence -> clarification flow
    if (intent.confidence < CONFIDENCE_THRESHOLD) {
      return {
        status: 'clarification',
        message: '无法理解你的意图。试试选择下方的推荐行动，或用 /help 查看命令。',
        candidates: [],
      };
    }

    // Convert intent to GameAction
    const action: GameAction = {
      type: intent.action,
      target: intent.target,
      modifiers: intent.modifiers ?? {},
      source: 'intent',
    };
    return { status: 'success', action };
  } catch {
    return {
      status: 'error',
      message: '无法理解你的意图。试试选择下方的推荐行动，或用 /help 查看命令。',
    };
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Player text input -> Command parser | Structured commands parsed by Commander.js; invalid input returns null |
| Player text input -> Intent classifier | Free-form NL sent to LLM API; response validated by Zod schema |
| LLM API response -> Game engine | LLM output NEVER reaches Rules Engine directly; Zod validation + confidence gating first |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-11 | Tampering | intent-classifier.ts | mitigate | LLM response validated against IntentSchema (Zod). Invalid responses trigger retry (D-26). After max retries, error returned to player — never passed to Rules Engine. |
| T-01-12 | Elevation of Privilege | input-router.ts | mitigate | D-25 validation chain: Intent Classifier -> Zod validation -> confidence gating -> GameAction. Rules Engine only receives validated GameAction objects. Raw LLM output never reaches adjudication. |
| T-01-13 | Spoofing | intent-classifier.ts | mitigate | System prompt is isolated from player input (separate fields in generateObject). Player text is placed in the `prompt` field, not concatenated with system instructions. |
| T-01-14 | Denial of Service | command-parser.ts | mitigate | Commander.js exitOverride() prevents process.exit. configureOutput suppresses stdout/stderr writes. Unknown commands return null gracefully. |
</threat_model>
