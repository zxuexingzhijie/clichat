```typescript
export const SuccessGradeSchema = z.enum([
  'critical_success', 'great_success', 'success',
  'partial_success', 'failure', 'critical_failure',
]);
export type SuccessGrade = z.infer<typeof SuccessGradeSchema>;

export const GRADE_LABELS: Record<SuccessGrade, string>;
export const ATTRIBUTE_LABELS: Record<AttributeName, string>;

export const CheckResultSchema = z.object({
  roll: z.number().int().min(1).max(20),
  attributeName: AttributeNameSchema,
  attributeModifier: z.number(),
  skillModifier: z.number(),
  environmentModifier: z.number(),
  total: z.number(),
  dc: z.number(),
  grade: SuccessGradeSchema,
  display: z.string(),
});
export type CheckResult = z.infer<typeof CheckResultSchema>;

export const DamageResultSchema = z.object({
  weaponBase: z.number(),
  attributeModifier: z.number(),
  gradeBonus: z.number(),
  armorReduction: z.number(),
  total: z.number(),
  display: z.string(),
});
export type DamageResult = z.infer<typeof DamageResultSchema>;
```

Grade thresholds:
  - nat20 -> critical_success (always, regardless of modifiers)
  - nat1 -> critical_failure (always, regardless of modifiers)
  - total >= DC + 10 -> great_success
  - total >= DC -> success
  - total >= DC - 5 -> partial_success
  - total < DC - 5 -> failure

D-19 Damage formula:
  total = weaponBase + attributeModifier + gradeBonus - armorReduction
  total = Math.max(0, total)  // no negative damage

D-20 Check display format:
  `[D20: ${roll}] + ${attrLabel} ${attrMod} = ${total} vs DC ${dc} -> ${gradeLabel}`

Create `src/engine/adjudication.ts`:
- `resolveNormalCheck(params: CheckParams)`: Computes total = roll + attrMod + skillMod + envMod. Nat20 -> critical_success, Nat1 -> critical_failure regardless of total. Then grade by thresholds. Builds display string per D-20 using ATTRIBUTE_LABELS and GRADE_LABELS from src/types/common.ts. Returns CheckResult.
- `resolveOpposedCheck(params: OpposedCheckParams, rng?)`: D-14 logic. Two rolls, compare totals. Tie -> defender wins. Returns { attacker: CheckResult, defender: CheckResult, winner: 'attacker' | 'defender' }.
- `resolveProbabilityCheck(params: ProbabilityCheckParams, rng?)`: D-15 logic. Percentage vs threshold. Returns { roll: number, threshold: number, success: boolean }.
- `resolvePlotCriticalCheck(params: CheckParams)`: D-16 logic. Same grading as normal, adds narrative_hint string suggesting AI narration tone based on grade.

- `calculateDamage(params: DamageParams)`: D-19 formula. total = weaponBase + attributeModifier + getGradeBonus(grade) - armorReduction. Floor at 0. Returns DamageResult.
