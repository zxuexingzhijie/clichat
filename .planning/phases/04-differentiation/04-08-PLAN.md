**1. Create `src/ui/components/diff-line.tsx` per UI-SPEC:**

```typescript
import React from 'react';
import { Box, Text } from 'ink';
import type { DiffMarker } from '../../engine/branch-diff';

type DiffLineProps = {
  readonly marker: DiffMarker;
  readonly content: string;
  readonly isHighImpact: boolean;
};

export function DiffLine({ marker, content, isHighImpact }: DiffLineProps): React.ReactNode {
  const markerColor = marker === '+' ? 'green' : marker === '-' ? 'red' : 'yellow';
  return (
    <Box flexDirection="row">
      <Text color={markerColor}>{marker} </Text>
      <Text>{content}</Text>
      {isHighImpact && <Text color="yellow" bold>  !高影响</Text>}
    </Box>
  );
}
```

```typescript
type BranchSaveInfo = {
  readonly saveId: string;
  readonly saveName: string;

  readonly questStage: string;

  readonly branchMeta: BranchMeta;
  readonly saves: readonly BranchSaveInfo[];
  readonly children: readonly BranchDisplayNode[];
};

type BranchTreePanelProps = {
  readonly tree: readonly BranchDisplayNode[];
  readonly currentBranchId: string;
  readonly onClose: () => void;
  readonly onCompare: (branchId: string) => void;
  readonly onSwitch: (branchId: string) => void;
};
```

Layout per UI-SPEC:
- Header: `<Text bold color="cyan">【分支树】</Text>` ... `<Text dimColor>Esc 返回</Text>`

- Tree rendering: recursive function that produces lines with Unicode box-drawing:
  ```
  main ──● save-01 (Day 1, 黑松镇)
    │    └── save-02 (Day 2, 幽暗森林)
    │
    ├── rescue ──● save-03 (Day 2, 矿洞)    ← current
    │            └── save-04 (Day 3, 黑松镇)
    │
    └── betray ──● save-05 (Day 2, 暗道)
  ```

- Tree line rendering:
  - Branch name: `<Text bold>{name}</Text>` if current, `<Text>{name}</Text>` otherwise
  - `──●` for HEAD save (headSaveId), `──○` for other saves
  - Current branch: append `← current` in cyan
  - `│` for continuation, `├──` for non-last child, `└──` for last child
  - Indent children by 2 spaces per level

- Node info per save: `{saveName} ({gameTime}, {location})`
- Selected node: entire line in bold cyan

- useInput: Up/Down to move selection cursor, Enter to view detail, `c` to compare (calls onCompare with selected branch), Esc to close

- **Wide (>= 100):** tree left 60%, detail pane right 40%. Detail shows: save name, game time, location, key quest stages (1-2 lines), NPC relationship changes (max 3), branch source.
- **Narrow (< 100):** full-width tree, Enter opens detail as overlay

- Empty state (only main, no other branches): `目前只有主线剧情。使用 /branch {name} 在关键抉择前创建分支，探索不同命运。`
- No saves on branch: `当前分支尚无存档。游戏会自动保存，或使用 /save 手动存档。`

- Hint bar: `<Text dimColor>↑↓ 选择节点    Enter 查看详情    c 对比    Esc 返回</Text>`
