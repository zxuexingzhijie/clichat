export function migrateV2ToV3(raw: unknown): unknown {
  if (typeof raw !== 'object' || raw === null) return raw;
  const data = raw as Record<string, unknown>;
  if (data['version'] !== 2) return raw;
  return {
    ...data,
    version: 3,
    branchId: 'main',
    parentSaveId: null,
    exploration: { locations: {} },
    playerKnowledge: { entries: {} },
    turnLog: [],
  };
}

Create `src/persistence/branch-manager.ts` following the save-file-manager.ts pattern (per D-01, D-02):

```typescript
import * as nodeFs from 'node:fs';
import path from 'node:path';
import { nanoid } from 'nanoid';
import { branchStore, type BranchMeta } from '../state/branch-store';
import { eventBus } from '../events/event-bus';

export const _fs = nodeFs;

function sanitizeBranchName(name: string): string {
  return name.replace(/[^a-zA-Z0-9\u4e00-\u9fff_-]/g, '-');
}

function guardPathTraversal(filePath: string, baseDir: string): void {
  const resolved = path.resolve(filePath);
  const resolvedBase = path.resolve(baseDir);
  if (!resolved.startsWith(resolvedBase + path.sep) && resolved !== resolvedBase) {
    throw new Error(`Path traversal detected: ${filePath} is outside save directory`);
  }
}
```

Implement these exported functions:

**`createBranch(name: string, description?: string): BranchMeta`**
- Sanitize name
- Generate id with `nanoid()`
- Get current branch state
- Create `BranchMeta` with `parentBranchId: currentBranchId`, `parentSaveId: current branch's headSaveId`, `headSaveId: null`, `createdAt: new Date().toISOString()`
- Update branchStore: add branch to `branches` record, set `currentBranchId` to new id
- Return the new BranchMeta

**`switchBranch(branchId: string): void`**
- Validate branchId exists in branchStore
- Throw `Error('分支不存在')` if not found
- Update branchStore: set `currentBranchId` to branchId

**`deleteBranch(branchId: string): void`**
- Validate branchId exists
- Throw if branchId === currentBranchId: `Error('无法删除当前所在分支')`
- Remove branch from branchStore `branches` record

**`listBranches(): readonly BranchMeta[]`**
- Return `Object.values(branchStore.getState().branches)`

**`getBranchTree(): BranchTreeNode[]`**
- Build tree structure from branch registry
- `type BranchTreeNode = { readonly branch: BranchMeta; readonly children: readonly BranchTreeNode[] }`
- Root is the branch with `parentBranchId === null` (main)
- Children are branches whose `parentBranchId` matches parent's id

**`updateBranchHead(branchId: string, saveId: string): void`**
- Update `headSaveId` for the given branch in branchStore

**`saveBranchRegistry(saveDir: string): Promise<void>`**
- Write `branches.json` to saveDir with `Bun.write()`
- Content: `JSON.stringify(branchStore.getState())`
- Guard path traversal

**`loadBranchRegistry(saveDir: string): Promise<void>`**
- Read `branches.json` from saveDir with `Bun.file().text()`
- Parse with `BranchStateSchema.safeParse()`
- Restore to branchStore via `setState`
- If file doesn't exist, keep default state (main branch)

Export `BranchTreeNode` type.
