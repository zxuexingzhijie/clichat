    ```typescript
    const config = getRoleConfig('narrative-director');
    const system = buildNarrativeSystemPrompt(context.sceneType);
    const dynamicPrompt = buildNarrativeUserPrompt(context as NarrativeUserPromptContext);

    let textResult: string;
    let usageResult: { inputTokens: number; outputTokens: number; totalTokens: number };

    if (config.providerName === 'anthropic') {
      // Anthropic: messages format with ephemeral cacheControl on static system content
      const { text, usage } = await generateText({
        model: config.model(),

        maxTokens: config.maxTokens,
        messages: [

            content: [
              {
                type: 'text',
                text: system,
                providerOptions: {
                  anthropic: { cacheControl: { type: 'ephemeral' } },
                },
              },
              { type: 'text', text: dynamicPrompt },
            ],
          },
        ],
      });
      textResult = text;
      usageResult = usage;
    } else {
      // Google/OpenAI: system + prompt shorthand; Google auto-caches stable prefix
      const { text, usage } = await generateText({
        model: config.model(),
        temperature: config.temperature,
        maxTokens: config.maxTokens,
        system,   // static — same value per sceneType across turns = Google caches automatically
        prompt: dynamicPrompt,
      });
      textResult = text;
      usageResult = usage;
    }
    recordUsage('narrative-director', usageResult);
    ```
