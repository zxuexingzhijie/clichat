```yaml
name: CI

on:
  pull_request:
    branches: [main]
  push:
    branches: [main]

permissions:
  contents: read

jobs:
  quality-gate:
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - uses: actions/checkout@v4

      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: "1.3.12"

      - name: Install dependencies
        run: bun install --frozen-lockfile

      - name: Type check
        run: bun run typecheck

      - name: Test
        run: bun test

      - name: Build
        run: bun run build
```

Key points:
- `permissions: contents: read` follows least-privilege principle
- `--frozen-lockfile` ensures reproducible installs
- Runs typecheck BEFORE test (fast fail on type errors)
- Includes build step to catch bundle errors early
- `timeout-minutes: 10` prevents runaway jobs

Create `.github/workflows/release.yml` per D-10, D-11, D-12, D-13 (release pipeline sequence: test -> build -> npm publish -> binary build x3 -> GitHub Release -> Homebrew formula update).

```yaml
name: Release

on:
  push:
    tags: ['v*']

permissions:
  contents: write

jobs:
  quality-gate:
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: "1.3.12"
      - run: bun install --frozen-lockfile
      - run: bun run typecheck
      - run: bun test

  publish-npm:
    needs: quality-gate
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: "1.3.12"
      - run: bun install --frozen-lockfile
      - run: bun run build
      - run: bun publish --access public
        env:
          NPM_TOKEN: ${{ secrets.NPM_TOKEN }}

  build-binaries:
    needs: quality-gate
    runs-on: ubuntu-latest
    timeout-minutes: 15
    strategy:
      matrix:
        include:
          - target: bun-darwin-arm64
            artifact: chronicle-darwin-arm64
          - target: bun-darwin-x64
            artifact: chronicle-darwin-x64
          - target: bun-linux-x64
            artifact: chronicle-linux-x64
    steps:
      - uses: actions/checkout@v4
      - uses: oven-sh/setup-bun@v2
        with:
          bun-version: "1.3.12"
      - run: bun install --frozen-lockfile

      - name: Compile standalone binary
        run: |
          bun build --compile src/cli.ts \
            --target ${{ matrix.target }} \
            --outfile ${{ matrix.artifact }} \
            --minify \
            --external react-devtools-core \
            ./world-data/**/*.yaml \
            ./world-data/*.yaml \
            ./world-data/world-manifest.json

      - name: Create tarball
        run: tar czf ${{ matrix.artifact }}.tar.gz ${{ matrix.artifact }}

      - name: Compute SHA256
        id: sha
        run: echo "sha256=$(shasum -a 256 ${{ matrix.artifact }}.tar.gz | cut -d ' ' -f1)" >> "$GITHUB_OUTPUT"

      - uses: actions/upload-artifact@v4
        with:
          name: ${{ matrix.artifact }}
          path: ${{ matrix.artifact }}.tar.gz

  release:
    needs: [publish-npm, build-binaries]
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - uses: actions/download-artifact@v4
        with:
          path: artifacts
          merge-multiple: true

      - name: Create GitHub Release
        uses: softprops/action-gh-release@v2
        with:
          generate_release_notes: true
          files: artifacts/*.tar.gz
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

  update-homebrew:
    needs: release
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - uses: actions/download-artifact@v4
        with:
          path: artifacts
          merge-multiple: true

      - name: Compute SHA256 hashes
        id: hashes
        run: |
          echo "darwin_arm64=$(shasum -a 256 artifacts/chronicle-darwin-arm64.tar.gz | cut -d ' ' -f1)" >> "$GITHUB_OUTPUT"
          echo "darwin_x64=$(shasum -a 256 artifacts/chronicle-darwin-x64.tar.gz | cut -d ' ' -f1)" >> "$GITHUB_OUTPUT"
          echo "linux_x64=$(shasum -a 256 artifacts/chronicle-linux-x64.tar.gz | cut -d ' ' -f1)" >> "$GITHUB_OUTPUT"

      - name: Update Homebrew formula
        uses: peter-evans/repository-dispatch@v3
        with:
          token: ${{ secrets.HOMEBREW_TAP_TOKEN }}
          repository: OWNER/homebrew-chronicle
          event-type: update-formula
          client-payload: >-
            {
              "version": "${{ github.ref_name }}",
              "darwin_arm64_sha256": "${{ steps.hashes.outputs.darwin_arm64 }}",
              "darwin_x64_sha256": "${{ steps.hashes.outputs.darwin_x64 }}",
              "linux_x64_sha256": "${{ steps.hashes.outputs.linux_x64 }}"
            }
```

Key design decisions:
- `publish-npm` and `build-binaries` run in PARALLEL after quality-gate (both depend only on quality-gate, not each other)
- `release` waits for BOTH npm publish and binary build to complete
- `update-homebrew` uses `repository_dispatch` to trigger formula update in the separate `homebrew-chronicle` repo
- Matrix strategy builds all 3 platforms in parallel on a single Linux runner using Bun cross-compilation
- `--external react-devtools-core` prevents the known build failure (Pitfall 2 from RESEARCH.md)
- World data files embedded via glob patterns in the compile command
- SHA256 computed per-artifact for Homebrew formula integrity
- `HOMEBREW_TAP_TOKEN` is a PAT with repo scope for the homebrew-chronicle repo
- `OWNER` placeholder appears in repository dispatch target -- user replaces during setup

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions -> npm registry | NPM_TOKEN secret used for publishing |
| GitHub Actions -> homebrew-chronicle repo | HOMEBREW_TAP_TOKEN PAT for cross-repo dispatch |
| Tag push -> workflow trigger | Any tag matching v* triggers release |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-07 | Information Disclosure | NPM_TOKEN leak | mitigate | Stored as GitHub Actions encrypted secret; never logged; `--access public` is explicit (no accidental private publish) |
| T-10-08 | Spoofing | Malicious tag push | mitigate | Quality gate runs full test suite before any publish; branch protection on main prevents unauthorized tags |
| T-10-09 | Tampering | Binary integrity | mitigate | SHA256 checksums computed in CI and embedded in Homebrew formula; users verify via `brew audit` |
| T-10-10 | Information Disclosure | HOMEBREW_TAP_TOKEN scope | mitigate | PAT scoped to only the homebrew-chronicle repository; minimum permissions needed |
</threat_model>
